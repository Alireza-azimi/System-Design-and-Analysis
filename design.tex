	\chapter{مقدمه}
شرکت سریع ترابر ماهان (ماهکس) با بیش از یک دهه تجربه درخشان،  خدمات پستی متوعی را با بالاترین استاندارد های کیفیت و سرعت  به مشتریان خود ارائه می دهد. این شرکت با تکیه بر ناوگان پیشرفته و سیستم های مدیریت هوشمند، خدماتی نظیر پست داخلی ، درون شهری و برون شهری ، کارگو و پست سریع (فلش) را در سرتاسر نقاط ایران را  فراهم کرده تا نیاز های گوناگون مشتریان خود را پوشش دهد.

ماهکس با ارائه خدماتی مطمئن سریع و هماهنگ با نیاز های روز، تجربه ای  بی نقص و اسوده را در ارسال مرسولات به ارمغان می آورد.

ماهکس علاوه بر خدمات پستی استاندارد، با ارائه راهکار های ویژه ای همچپون سفارش گردانی، انباردرای و بسته بندی، فرآنید ارسال و دریافت کالا را برای خود و مشتریان خود ساده و کارآمد کرده است.

این خدمات با امکاناتی نظیر پرداخت در محل (\lr{COD})، پسکرایه و پوشش بیمه ای جامع همراه هستندکه امنیت و اطمینان را برای تمامی مرسولات فراهم می کنند. همچنین ماهکس با امکان سفارشی سازی خدمات برای کسب و کار ها و ارائه پشتیبانی مستمر همواره در کنار مشتریان خود است.\footnote{وب‌سایت رسمی شرکت ماهکس}

همکاری با شرکت‌های مطرحی مانند دیجی‌کالا ، اسنپ و تکنولایف موجب شده است ماهکس سهم قابل‌توجهی از بازار حمل‌ونقل سریع کشور را در اختیار بگیرد. این همکاری‌ها در بهبود فرآیندهای توزیع، افزایش دقت در رهگیری مرسوله‌ها و ارتقای رضایت مشتریان نقش بسزایی داشته‌اند.

به صورت کلی خدمات پستی که شرکت ماهکس ارائه می دهد به صورت زیر می باشد:
\begin{itemize}
	\item 
پست زمینی : شرکت ماهکس با دردست داشتن یکی از بزرگترین ناوگان های حمل و نقل زمینی هر روزه و به طور مستمر به سرتاسر مرکز استان های ایران ارسال زمینی دارد.
	\item 
پست هوایی :شرکت ماهکس با داشتن دفاتر بار هوایی در تمامی فرودگاه های پر تعدد ایران هر روزه به ارسال ، دریافت و پذیرش بار در این دفاتر می پردازد.
	\item 
سایر خدمات: خدماتی نظیر پس کرایه،بسته بندی، ارسال درب به درب ،داشتن شعب در تمامی مراکز استان ها و  نمایندگی های فعال (بیش از ۳۰۰ نمایندگی فعال در سرتاسر ایران) همواره در تلاش است تا خدمات خود را بهتر و تخصصی تر ارائه دهد.
\end{itemize}


\section{شناخت سیستم های موجود}
با توجه به بزرگ مقباس بودن شرکت و داشتن کاربران بسیار زیادی که هروزه از این سیستم استفاده می کنند، این شرکت دارای ۴ سیستم اصلی برای انجام امور خود می باشد که به صورت زیر است.
\begin{enumerate}
	\item سیستم \lr{e-courier}

سامانه ای جامع  برای انجام کلیه ای امور پستی از جمله وظایف این سیستم می توان به موارد زیر اشاره نمود:
\begin{itemize}
	\item انجام کلیه امور ثبت و صدور بارنامه 
	\item مدیریت انبار و انبارداری
	\item ثبت وضعیت مرسوله ها و تخصیص مرسولات به پیک ها
\end{itemize}
	\item سیستم مدیریت فرآیند
	
	سیستمی درون سازمانی است که برای انجام امور زیر توسعه گردیده است:
\begin{itemize}
\item  تمامی امرور مربوط به پرسنل از جمله مرخصی در خواست مرخصی ، وام یا....
\item تغییر یا حذف اطلاعات درون سامانه \lr{e-courier} مستقیما از این سامانه انجام می شود. (مثل حذف یا تغییر بارنامه و...)
\end{itemize}
	\item سایت اصلی شرکت \footnote{\url{https://mahex.com}}
	
سایت اصلی شرکت برای معرفی شرکت ، معرفی خدمات و .... تعبیه گردیده است.
\begin{itemize}
	\item معرفی شرکت و خدمات آن
	\item امکان پیگیری مرسولات  
	\item محاسبه قیمت  براساس وزن ، مبدا و مقصد و...
	\item پشتیبانی 
\end{itemize}
	\item اپلیکیشن موبایل

اپلیکیشن موبایل برای تیم عملیات توسعه داده شده، تا پیک ها بتوانند به راحتی تعداد مرسولاتی که دارد را مشاهده کرده و به راحتی بتواند وضعیت هر مرسوله را در آن ثبت کننده. (به منظور تسریع زمان ثبت وضعیت مرسوله ها و مدیریت بسته های موجود توسط پیک)
\end{enumerate}

\section{ معایب سیستم موجود}

با توجه به ۴ سیستم موجود و دانستن وابستگی های هر سیستم به یکدیگر، معایب این سیستم به شرح زیر خواهد بود:

\begin{enumerate}
	\item نبود یا ضعف در مقیاس پذیری: کند شدن سایت در پیک ، قعطی های موقت زیاد ، صف طولانی در پردازش مرسولات
	\item یکپارچگی ضعیف  سیستم ها: ها تبادل دستی  داده بین سیستم ها ، خروجی و ورودی ناسازگار و...
	\item اتوماسیون بسیار کم است : فرآیند دستی و کاغذ بازی بسیار زیاد
	\item مسیر ها و جداول زمانی ناوگان ناکارامد
	\item مدیریت نا مناسب بازاریابی / قیمت گذاری و عدم شفافیت هزینه ها
	\item ضعف های تجربه کاربری \lr{UX} و تعامل با  مشتری
	\item پشتیبانی  ضعیف
	\item گزارش دهی بسیار ضعیف و بی درنگ نبودن اطلاعات مدیریت
	\item عدم شفافیت در روند بیمه و مسئولیت خسارت 
	\item عدم وجود سیستمی مبتنی بر هوش تجاری  
	\item وجود مشکلات متعدد در سیستم گزارش گیری از سامانه ها مخصوصا در امور حسابداری
\end{enumerate}
این موارد تنها بخشی از مشکلات سیستم موجود در شرکت سریع ترابر ماهان است.

\subsection{بررسی ساختار شرکت ماهکس}
شرکت سریع ترابر ماهان با نام تجاری ماهکس در سال ۱۳۹۰ با هدف ارائه راهکارهای مفید و مؤثر در حوزه صنعت لجستیک کشور تأسیس شد. این شرکت، که مالکیت آن متعلق به هواپیمایی ماهان است، بر پایه نیازهای مشتریان بنا نهاده شده و بر سرعت، دقت و بهینه‌سازی فرآیندهای ارسال تمرکز دارد. ماهکس به عنوان یک شرکت لجستیک فعالیت می‌کند و خدمات خود را بر اساس اصل مشتری‌مداری و استفاده از فناوری‌های نوین ارائه می‌دهد.

ساختار ماهکس شامل دفتر مرکزی در تهران (بزرگراه فتح، پلاک ۱۲۶) و شبکه‌ای گسترده است: بیش از ۳۱ شعبه فعال، بیش از ۳۰۰ نمایندگی  و پوشش بیش از ۹۰۰ تا ۲۵۰۰ شهر در سراسر ایران. این شبکه با موقعیت استراتژیک شعب (نزدیکی به فرودگاه‌ها، راه‌آهن و جاده‌های اصلی) امکان حمل‌ونقل چندوجهی (زمینی، هوایی) را فراهم می‌کند. شرکت همچنین دارای ناوگان ملکی برای برنامه‌ریزی سفارشی ارسال‌ها در هر ساعت از شبانه‌روز است.

ماهکس خدمات گسترده‌ای در حوزه حمل‌ونقل پستی و لجستیک ارائه می‌دهد، از جمله:
\begin{itemize}
	\item پست درون‌شهری و بین‌شهری: ارسال سریع و اقتصادی مرسولات با رهگیری ۲۴ ساعته.
	\item پست هوایی داخلی: استفاده از ناوگان هواپیمایی ماهان برای ارسال سریع.
	\item
خدمات ویژه: پس‌کرایه، پرداخت در محل (\lr{COD})، بسته‌بندی حرفه‌ای، بیمه مرسولات، و فولفیلمنت (مدیریت انبار و سفارش‌ها). 
	\item 
حمل درب‌به‌درب: همه مرسولات بدون هیچ هزینه اضافه ای در مبدا دریافت در مقصد تحویل داده می شوند.
\end{itemize}
این خدمات با تمرکز بر صنایع مختلف (مانند تجارت الکترونیک) و مدیریت لجستیک معکوس (بازگشت کالا) ارائه می‌شوند.

\subsection{شناسایی مشکلات، نیازها و تحلیل محیطی}
مشکلات اصلی که شرکت ماهکس با آن مواجه هستش :
\begin{enumerate}
	\item 
	چالش‌های عملیاتی در رهگیری و مدیریت: تأخیر در رهگیری آنلاین به دلیل سیستم‌های غیریکپارچه، که منجر به نارضایتی مشتریان و خسارت مالی می‌شود (مانند مفقودی کالا با سقف خسارت ۹۰\% ارزش).
	\item 
امنیت و بسته‌بندی: ارسال اقلام حساس (جواهرات، مدارک) با ریسک توقیف یا آسیب، به دلیل عدم آگاهی از قوانین و لیست ممنوعات.
	\item 
عدم یکپارچگی سیستم‌ها: سیستم‌های جداگانه برای پنل، وب‌سرویس و گزارش‌گیری، که نیاز به پلتفرم \lr{ERP} یکپارچه دارد.
	\item 
	ریسک‌های خارجی: هزینه‌های بالای حمل (تا ۵۰\% ارزش کالا)، کمبود راننده، تحریم‌ها، عدم توسعه مراکز لجستیکی و نبود حمل چندوجهی کارآمد.
	\item 
	چالش‌های دیجیتال: وابستگی به فرآیندهای دستی ، که با رشد بازار نیاز به \lr{AI} و \lr{GPS} برای ردیابی واقعی‌زمان را افزایش می‌دهد.
\end{enumerate}

نیاز های اصلی شرکت ماهکس  :
\begin{enumerate}
	\item 
دیجیتال‌سازی کامل فرآیندها برای کاهش تأخیرها.
	\item 
ادغام فناوری‌های نوین مانند \lr{IoT} برای مدیریت انبار، ناوگان و....
	\item 
آموزش کارکنان برای پذیرش سیستم‌های جدید.
	\item 
عدم وجود داشبورد مدیریتی کارآمد و دقیق و نبود ابزار های \lr{BI}
\end{enumerate}

 نقاط قوت، نفاط ضعف، فرصت ها وتهدید ها نیز به صورت روبرو است.
\begin{itemize}
	\item نقاط قوت:
	
شبکه گسترده (۳۱ شعبه + ۳۰۰ نمایندگی، پوشش ۹۰۰+ شهر)، ناوگان ملکی و هوایی ماهان، خدمات متنوع، مراکز پردازش پیشرفته، موقعیت استراتژیک شعب، و تمرکز بر مشتری‌مداری.
	\item نقاط ضعف:
	
	(\lr{Weaknesses})سیستم‌های اطلاعاتی قدیمی و غیریکپارچه، چالش‌های امنیتی در ارسال حساس، مدیریت ضعیف لجستیک معکوس (بازگشت کالا)، وابستگی به فرآیندهای دستی، و ریسک خسارت/تأخیر
	\item فرصت‌ها:
	
رشد تجارت الکترونیک (تقاضا برای فولفیلمنت)، ادغام با پلتفرم‌های مانند دیجی‌کالا، گسترش به بازارهای جهانی، استفاده از فناوری‌های نوین (\lr{AI، cloud)}، و کمپین‌های اجتماعی برای برندینگ
	\item تهدیدها:

رقابت شدید (پست جمهوری اسلامی، تیپاکس)، تغییرات گمرکی و تحریم‌ها، هزینه‌های بالای حمل، کمبود راننده، و عدم توسعه زیرساخت‌های لجستیکی ملی.
\end{itemize}

\subsection{امکان سنجی}
\begin{itemize}
	\item امکان سنجی فنی
\begin{itemize}
	\item 
وضعیت فعلی: استفاده از پنل و وب‌سرویس موجود، اما نیاز به ارتقا به سیستم‌های \lr{cloud-based} (مانند \lr{AWS}) و ادغام \lr{GPS/IoT }برای ردیابی واقعی‌زمان.
	\item 
توسعه \lr{ERP} یکپارچه با فناوری‌های \lr{Java} (\lr{backend})، \lr{React} (\lr{frontend}) و \lr{API}های خارجی برای پرداخت و رهگیری.
	\item 
منابع: دسترسی به توسعه‌دهندگان (هزینه متوسط ۵۰ میلیون تومان/ماه برای تیم ۵ نفره).
	\item 
ارزیابی: کاملاً امکان‌پذیر؛ فناوری‌های مورد نیاز در دسترس است و می‌توان از مدل \lr{SaaS} برای کاهش هزینه استفاده کرد.
\end{itemize}
	\item امکان‌سنجی اقتصادی 
	\begin{itemize}
		\item تحلیل هزینه-فایده (\lr{CBA}): هزینه‌ها: توسعه نرم‌افزار (۵۰۰ میلیون تومان)، سخت‌افزار/سرور (۳۰۰ میلیون تومان)، آموزش (۱۵۰ میلیون تومان)، مجموع اولیه ۹۵۰ میلیون تومان. فواید: کاهش ۲۵-۳۵ \% تأخیرها (صرفه‌جویی ۱.۵ میلیارد تومان سالانه)، افزایش ۲۰\% درآمد از مشتریان جدید (۳ میلیارد تومان)، و کاهش خسارت (۵۰۰ میلیون تومان).
		\item 
	\lr{ROI} و \lr{NPV}: بازگشت سرمایه در ۱۵-۲۰ ماه، با نرخ سود ۳۰\% (\lr{NPV} مثبت ۲ میلیارد تومان در ۳ سال).
		\item ارزیابی: اقتصادی و سودآور؛ با توجه به رشد بازار لجستیک.
	\end{itemize}
	\item امکان‌سنجی عملیاتی
	\begin{itemize}
		\item تأثیر بر کاربران: آموزش ۴-۶ هفته‌ای برای ۲۰۰ کارمند، کاهش زمان ثبت سفارش از ۱۵ به ۳ دقیقه.
		\item پذیرش: فرهنگ نوآورانه شرکت (با پشتوانه ماهان) پذیرش را تسهیل می‌کند، اما مقاومت اولیه ممکن است.
		\item ارزیابی: با توجه به حجم نیرو ها و ساختار شرکت شدنی اما زمان بر است.
	\end{itemize}
	\item امکان‌سنجی زمان‌بندی		

زمان‌بندی تقریبی: یک سال الی یک سال و دو ماه

فاز 1 (شناخت نیاز ها) : 1 الی 1/5 ماه

فاز 2 (تحلیل دقیق تر سیستم و نیازمندی ها): 1 الی 2 ماه

فاز 3 (طراحی و ساخت سیستم):  10 ماه

فاز 4 (تحویل به مشتری و نگهداری): تا یک سال بعد از اتمام پروژه

\end{itemize}

\section{فاز ۲: نیازمندی‌ها و تحلیل دقیق} 
\subsection{مشکلات سیستم}
با توجه به بزرگ مقیاس بودن پروژه و با توجه به این موضوع که باید تمام تغییرات  روی سرس کد اصلی اجرا شوند لذا لیستی از تمام مشکلاتی که با ان مواجه هستیم به صورت زیر خواهد بود:

\begin{enumerate}
	\item ضعف های فنی و معماری سیستم ها 
	\begin{itemize}
		\item تغییرات کوچک نیاز به اصلاح بخش های زیادی از کد دارد .
\begin{itemize}
	\item 
ریشه ها : طراحی اولیه تک کدی، استفاده از تکلونوژی های قدیمی یا تکیه به نسخه های قدیمی کتابخانه ها.
	\item تاثیر: کندی سیستم توسعه، هزینه بالای نگهداری، ریسک بالای باگ در تغییرات.
	\item راهکار: مهاجرت به معماری \lr{MicroService}
\end{itemize} 
\item نبود یا ضعف در مقیاس پذیری
\begin{itemize}
\item علائم:
کند شدن سایت در پیک، قطعی های موقت۷ صف طولانی در پردازی مرسولات
\item
ریشه: نبود \lr{load balancing}، نگهداری \lr{state} در سرور های تکی، طراحی دیتابیس ناکافی برای رشد.
\end{itemize} 
\item یکپارچگی ضعیف با سیستم خارجی 
\begin{itemize}
	\item 
تبادل دستی داده ها بین سیستم ، خرپجی و ورودی های ناسازگار، خطا های مکرر در \lr{import , export}.
	\item 
	استاندار های نامشخص ، نداشتن \lr{API} مستند و قابل اعتماد ، دیتا فرمت های متفاوت \lr{excel} که باعث تاخیر در عملیات ، خطای انسانی و ناسازگاری داده ها شده است.
\end{itemize}
	\end{itemize}
	\item ضعف های عملکردی و فرآیندی
\begin{itemize}
	\item فرآیند دستی و کاغذ بازی زیاد
	\begin{itemize}
		\item علائم : 
عملیات های روزانه نیازمند فرم های فیزیک یا اکسل محور و ورود دوباره این اطلاعات به سیستم است.
		\item ریشه ها :
نبود اتوماسیون ، مقاوت در برابر تغییر.
		\item  تاثیر ها:
خطای انسانی، کندی پردازش، هزینه های نیروی انسانی بالا.
		\item راهکار ها:
	خودکار سازی فرآیندهای تکراری، ایجاد فرم های دیجیتال.
	\end{itemize}
	\item مسیر ها و جداول زمانی  ناوگان ناکارا
	\begin{itemize}
		\item علائم : 
مسیر های طولانی تر از حد لازم۷ سوخت مصرفی بالا، تاخیر در تحویل.
		\item ریشه ها:
نبود مسیریاب هوشمند یا الگوریتم تخصیص دینامیک، عدم استفاده از داده های ترافیک زنده.
		\item تاثیر:
هزینه عملیاتی بالا، کاهش رضایت مشتری.
		\item راهکار ها:
استفاده از الگورتیم های مسیریابی، یکپارچه سازی با سرویس های ترافیک و نقشه ، تخصیص پویا مرسولات براساس الویت.
	\end{itemize}
		\item نگهداری ناوگان و برنامه ریزی سرویس های تعمیراتی ضعیف
	\begin{itemize}
		\item  علائم: 
	خرابی ناگهانی خودرو ها۷ عدم ثبت سرویس ها، هزینه تعمیراتی بالا.
		\item ریشه ها :
	نبود ماژول نگهداری، عدم نظارت بر ساعت عملکرد.
		\item تاثیر: 
افزایش هزینه های تعمیر، تاخیر و ریسک ایمنی.
		\item راهکار ها :
پیاده سازی نرم افزار ناوگان یا یک بخش درون اپلیکیشن برای راننده های ناوگان که اطلاعات سرویس ها و... در آن قرار داد و استفاده از \lr{IOT} برای مانیتورینگ سلامت خودرو.
	\end{itemize}
		\item مدیریت نامناسب بازاریابی / قیمت گذاری و عدم شفافیت هزینه ها
	\begin{itemize}
		\item علائم :
اختلاف در بین تعرفه ها در بین شعب، شکایت مربوط به هزینه های نامشخص.
		\item ریشته ها:
قیمت گذاری غیر شفاش
		\item تاثیر: 
نارضایتی  مشتری، و از دست دادن مشتریان بزرگ.
		\item راهکار ها:
موتور قیمت گذاری مرکزی، سیاست های شفاف، نمایش قیمت در زمان ثبت سفارش.
	\end{itemize}
\end{itemize}
\item ضعف های مربوط به نیروی انسانی و سازمانی 
\begin{itemize}
	\item مقاومت در برابر تغییر فرهنگ سازمانی سنتی  
	\begin{itemize}
		\item  علائم : 
مخالفت سیستم با خودکار سازی، اجرای ناقص \lr{SOP} ها.
		\item ریشه ها:
آموزش ناکافی، ترس از تغییر، نبود مدیریت تغییر.
		\item تاثیر ها:
شکست پروژه های فناوری
		\item راهکار ها :
برنامه مدیریت تغییر، جلسات آموزشی.
	\end{itemize}
		\item تیم \lr{IT} ضعیف
	\begin{itemize}
		\item  علائم:
		جذب نیروی متخصص ، پرداخت مناسب و نبود \lr{roadmap} فنی.
		\item ریشه ها:
کیفیت پایین توسعه و ریسک فنی.
		\item راهکار ها:
سرمایه گذاری در استخدام، قرارداد با شرکت های فنی برای پشتیبانی، آموزش داخلی
	\end{itemize}

\end{itemize}
\item ضعف های تجربه کاربری \lr{UX} و تعامل با مشتری
\begin{itemize}
	\item رابط کاربری نامناسب و یا پیچیده برای مشتریان و کاربران داخلی
	\begin{itemize}
		\item علائم:
نرخ ترک صفحه ثبت سفارش بالا، تماس های مکرر به پشتیبانی برای ثبت 
		\item ریشه ها:
فرم های طولانی۷ \lr{UI} غیر قابل فهم ، عدم دسترسی موبایل فرندلی به سامانه ها.
		\item تاثیر: 
	هزینه بالاتر پشتیبانی. 
		\item راهکار ها:
بازطراحی \lr{UX} با اصول ساده ، تست با کاربران واقعی، طراحی رسپانسیو.
	\end{itemize}
	\newpage
	\item پشتیبانی مشتری ضعیف یا پراکسی محور
	\begin{itemize}
		\item علائم:
زمان پاسخ دهی طولانی، رضایت پایین در تماس ها،  فقدان کانال های متنوع (چت، تیکت، و...)
		\item  ریشه ها:
نبود \lr{CRM} یکپارچه، آموزش ناکافی اپراتور ها.
		\item تاثیر :
از دست دادن مشتریان، شکایت عمومی .
		\item راهکار:
استقرار یک \lr{CRM} مرکزی متمرکز با تاریخچه کامل سفارشات، آموزش کامل تیم پشتیبانی.
	\end{itemize}
	\item گزارش دهی ضعیف و بی درنگ نبودن اطلاعات مدیریت
	\begin{itemize}
		\item علائم :
مدیران منتظر گزارش های روزانه و هفتگی هستند، تطبیق ناپذیری آماری
		\item ریشه ها :
نبود داشبورد \lr{real-time} و ابزار های \lr{BI} وجود ندارد.
		\item تاثیر: 
واکنش کند به مسائل عملیاتی، فرصت های از دست رفته بهینه سازی.
		\item راهکار:
		ایجاد یک داشبورد  \lr{BI} درون سیستم.
	\end{itemize}
\end{itemize}
\item ضعف های مربوط به امنیت عملیاتی و مطابقت با قوانین
\begin{itemize}
	\item عدم رعایت الزامات حریم خصوصی و قوانین
	\begin{itemize}
		\item علائم:
ذخیره سازی اطلاعات حساس بدون مراقبت ، قرار داد های ناقص با طرف سوم.
		\item ریشه ها:
عدم توجه به مقررات و نبود سیاست های داخلی.
		\item تاثیر ها:
ریسک جریمه و لطمه به برند.
		\item راهکار ها:
تدوین سیاست های حریم خصوصی ، قرار داد های \lr{DPA} یا شرکا و ممیزی اینترنی.
	\end{itemize}
	\item فقدان برنامه بازیابی از حادثه و پایداری
	\begin{itemize}
		\item علائم:
نبود بکاپ منظم، قطع سرویس به مدت طولانی در زمان بروز مشکل.
		\item ریشه ها:
هزینه اولیه، غفلت برنامه ریزی.
		\item تاثیر:
توقف عملیات و ضرر مالی.
		\item راهکارها:
تست دوره ای بازیابی ، \lr{backup policy} و \lr{DR site}
	\end{itemize}
\end{itemize}
\item ضعف های کیفیت خدمات و اعتبار برند
\begin{itemize}
	\item تاخیر های مکرر در تحویل و خطا های عمومی 
	\begin{itemize}
		\item علائم:
	شکایات عمومی، بازگشت مرسوله، کاهش تکرار سفارش مشتریان.
		\item ریشه ها:
	مشکلات عملیاتی، سیستم ردیابی ناکافی، \lr{dispatch} نامناسب.
		\item راهکار ها:
		\lr{time to pickup and on-time delivery}
	\end{itemize}
	\item  عدم شفافیت در روند بیمه و مسئولیت خسارت
	\begin{itemize}
		\item علائم:
اختلاف با مشتریان درباره پوشش خسارت، فرآیند پیچیده ادعا.
		\item ریشه ها:
عدم خودکار سازی ادعا ها و اسناد پراکنده.
		\item راهکارها:
	فرم ادعای دیجیتال، روند اتوماسیون ادعا و \lr{dashboard} برای پیگیری.
	\end{itemize}
\end{itemize}
\end{enumerate}

\subsection{ایده هایی برای بهبود سیستم}
تمرکز اصلی این ایده ها بر روی افزایش سرعت عملیات، کاهش خطای انسانی، ارتقای رضایت مشتری، بهبود تصمیم گیری مدیریتی، استفاده از فناوری های جدید در جمع آوری ، رهگیری و تحویل مرسوله است.
\begin{enumerate}
	\item ایده های بهبود در ثبت سفارش و پذیرش بار
	
هدف: کاهش زمان ثبت، حذف ورودی ها تکراری و افزایش دقت و عملکرد است.
	\begin{itemize}
		\item سامانه یکپارچه پذیرش
		\begin{itemize}
			\item 
طراحی فرم دیجیتال ساده برای ثبت سفارش توسط مشتری یا کاربر داخلی.
			\item 
ارتباط مستقیم با پایگاه داده مشتریان و حدف ورود دوباره اطلاعات
		\end{itemize}
				\item ماژول قیمت گذاری خودکار
		\begin{itemize}
			\item 
محاسبه خودکار هزینه حمل بر اساس وزن، نوع محموله، مسافت و زمان تحویل و نوع حمل.
			\item نمایش هزینه نهایی به مشتری پیش از تایید سفارش.
			\item  استفاده از الگوریتم یادگیری ماشین ساده برای پیشنهاد تحفیف به مشتریان وفادار
		\end{itemize}
						\item اعتبار سنجی هوشمند اطلاعات ورودی
		\begin{itemize}
			\item 
بررسی صحت شماره تلفن، کد ملی و ادرس مقصد در لحظه ورود.
			\item 
جلوگیری از اشتباهات در داده های اولیه که منجر به تحویل نادرست می شوند.
		\end{itemize}
	\end{itemize}
	\item ایده های بهبود در رهگیری پایش مرسولات

هدف: افزایش شفافیت و اطلاع رسانی لحظه ای به مشتریان و مدیران.
\begin{itemize}
	\item سیستم رهگیری هوشمند چند سطحی
	\begin{itemize}
		\item 
ترکیب داده های \lr{GPS} کامیون ها، اسکن بارکد در مراکز پردازش و داده های تحویل راننده.
		\item 
برای یک سری کالا های خاص مثل طلا یا .... امکان رهگیری آنلاین داشته باشیم.				
	\end{itemize}
	\item رهگیری داخلی برای مدیران
	\begin{itemize}
		\item 
داشبوردی با نقشه زنده از ناوگان در سراسر کشور.
		\item 
وضعیت تاخیر ها* توقف های غیر مجاز یا نقاط گلوگاهی.
	\end{itemize}
		\item سیستم هشدار خودکار تاخیر
	\begin{itemize}
		\item 
	اگر محموله ای بیش از زمان استاندارد در یک ایستگاه ماند، هشدار به سرپرست ارسال گردد.
		\item 
در صورت تاخیر بیش از حد پیامی به مشتری ارسال شود.
	\end{itemize}
\end{itemize}
	\item ایده های بهبود در مدیریت ناوگان و مسیر ها
	
هدف : کاهش هزینه سوخت، زمان حمل و افزایش بهره وری وسایل نقلیه.
\begin{itemize}
	\item مسیر یابی خودکار
	\begin{itemize}
		\item 
استفاده از الگوریتم های بهینه سازی مسیر بر اساس حجم مرسولات و وضعیت ترافیک. 
		\item 
تخصیص بهینه راننده به مسیر.
	\end{itemize}
	\item سیستم مدیریت نگهداری خودرو ها
	\begin{itemize}
		\item 
ثبت خودکار کیلومتر کارکرد و زمان آخرین سرویس از طریق \lr{GPS} یا ورودی راننده.
		\item هشدار خودکار برای سیرویس یا بیمه در آستانه انقاضا.
	\end{itemize}
	\item سیستم سوخت و هزینه های سفر
	\begin{itemize}
		\item
هر راننده هنگام بازگشت هزینه سوخت را در سیستم ثبت می کند.
		\item 
تحلیل مصرف سوخت به تفکیک مسیر و نوع خودرو.
	\end{itemize}
	\item داشبورد عملکرد رانندگان
	\begin{itemize}
		\item 
شاخص هایی ماند میانگین تاخیر، مصرف سوخت و امتیاز رضایت مشتری
		\item 
پاداش برای رانندگان منظم و کارآمد
	\end{itemize}
\end{itemize}
	\item ایدههای بهبود در فرآیند پردازش بار در مراکز توزیع 

هدف: کاهش زمان توقف بسته ها، افزایش دقت در تفکیک و بارگیری
\begin{itemize}
	\item مدیریت خودکار الویت بارگیری
	\begin{itemize}
		\item الگوریتمی که ترتیب بارگیری کامیون ها را براساس مسیر و زمان تحویل مشخص کند.
	\end{itemize}
	\item داشبور کنترل موجودی در مراکز پردازش
	\begin{itemize}
		\item 
آمار لحظه ای از تعداد بسته ها در هر مرکز، زمان ماندگاری و وضعیت خروج.	
	\end{itemize}
\end{itemize}
	\item ایده های بهبود در مدیریت ارتباط با مشتریان

هدف: افزایش رضایت مشتری، کاهش تماس های تکراری و تقویت وفاداری.
\begin{itemize}
	\item سیستم تکینگ هوشمند پشتیبانی 
	\begin{itemize}
		\item 
هر تماس یا شکایت مشتری به صورت خودکار ثبت و به واحد مروبطه ارجاع داده می شود.
		\item 
		پیامک اطلاع رسانی از وضعیت رسیدگی .
	\end{itemize}
	\item پروفایل مشتری و سابقه سفارشات
	\begin{itemize}
		\item 
کارمند بتواند تاریخچه کامل تعاملات را ببیند.
		\item
شناسایی مشتریان \lr{VIP} برای سرویس سریع تر. 
	\end{itemize}
	\item سیستم امتیاز دهی رضایت از مشتری
	\begin{itemize}
		\item 
بعد از تحویل، لینک نظر خواهی ارسال شود.
		\item 
	تحلیل داده ها برای یافتن نقاط ضعف فرآیند.
	\end{itemize}
	\item باشگاه مشتریان وفادار
	\begin{itemize}
		\item 
ارائه تخفیف یا امتیاز برای مشتریانی که به صورت مکرر ارسال دارند.		
	\end{itemize}
\end{itemize}	
\item ایده های بهبود در گزارش گیری و تصمیم مدیرتی 

هدف: ایجاد دید جامع برای تصمیم گیری و پایش عملکرد سازمان.
		
\begin{itemize}
	\item 	داشبورد مدیریتی بلادرنگ
		\begin{itemize}
				\item 
آمار لحظه ای از تعداد سفارشات، وضعیت ناوگان، تاخیر ها و رضایت مشتری.
				\item 
نمودار های مقایسه ای روزانه، هفتگی  و ماهانه.
		\end{itemize}
			\item 	گزارش های تحلیلی هزینه و درآمد
		\begin{itemize}
			\item 
تحلیل هزینه به تفکیک مسیر، نوع بار و یا راننده.
			\item 
شناسایی مسیر ها (میزان مرسولات یا پرهزینه بودن) برای تصیمیم گیری در نرخ.
		\end{itemize}
			\item پیش بینی مرسولات و حجم کاری در آنیده با هوش مصنوعی
		\begin{itemize}
			\item 
بر اساس داده های فصلی، تقویم تعطیلات و رفتار مشتریان، حجم بار آینده پیش بینی شود.
		\end{itemize}
		
\end{itemize}
\item ایده های بهبود در امنیت و کنترل دسترسی

هدف: جلوگیری از دسترسی غیر مجاز، نشت اطلاعات و حفظ امنیت داده ها.

\begin{itemize}
	\item 	سیستم نقش محور

\subsection{\lr{Use Case}}
در زیر با توجه به ساختار پروژه 5 \lr{Use Case} که برای ما اهمیت بیشتر دارند معرفی خواهد شد.
\begin{itemize}
	\item 
ثبت سفارش ارسال

بازیگر: مشتری

توضیح:
در این \lr{Use Case}، مشتری اطلاعات مربوط به بسته، مبدا، مقصد و نوع سرویس ارسال را در سیستم ثبت می‌نماید. پس از ثبت اطلاعات، سفارش به‌صورت خودکار در سیستم ذخیره شده و جهت بررسی به واحد مربوطه ارسال می‌شود.
\item 
بررسی و تأیید سفارش

بازیگر: کارشناس پذیرش

توضیح:
کارشناس پذیرش سفارش‌های ثبت‌شده را بررسی کرده و در صورت صحت اطلاعات، آن‌ها را تأیید می‌نماید. در غیر این صورت، امکان اصلاح یا رد سفارش وجود دارد.
\item
محاسبه هزینه ارسال

بازیگر: سیستم

توضیح:
سیستم بر اساس اطلاعات سفارش شامل وزن، مقصد و نوع سرویس، هزینه ارسال را محاسبه کرده و نتیجه را جهت اطلاع به مشتری نمایش می‌دهد.
\item
پیگیری وضعیت مرسوله

بازیگر: مشتری

توضیح:
مشتری می‌تواند با استفاده از کد رهگیری، وضعیت مرسوله خود را در مراحل مختلف ارسال از طریق سیستم مشاهده نماید.
\item 
مشاهده گزارش‌های مدیریتی

بازیگر: مدیر سیستم

توضیح:
مدیر سیستم قادر است گزارش‌های مختلفی از جمله تعداد سفارش‌ها، وضعیت ارسال‌ها و عملکرد کلی سیستم را در بازه‌های زمانی مشخص مشاهده نماید.
\end{itemize}

\section{فاز ۳ : طراحی و ساخت سیستم}
\begin{enumerate}
	\item 
برنامه‌ریزی ساخت سیستم

در این فاز، ابتدا برنامه‌ای دقیق برای ساخت سیستم تهیه شد تا اطمینان حاصل شود که تمام نیازمندی‌های شناسایی‌شده در فازهای قبلی به درستی اجرا شوند. برنامه شامل زمان‌بندی فعالیت‌ها، منابع انسانی (مانند تیم توسعه و تست) و اولویت‌بندی بخش‌های کلیدی سیستم مانند مدیریت فروش و انبار بود. هدف اصلی، تبدیل طراحی‌های تئوری به یک سیستم عملی بود که فرآیندهای شرکت ماهکس را بهینه کند، مانند کاهش زمان ثبت سفارشات از یک رب دقیقه به کمتر از ۵ دقیقه.


	\item اجرای نیازمندی‌های کارکردی
	
نیازمندی‌های کارکردی (مانند ثبت سفارش، به‌روزرسانی موجودی و تولید گزارش) در این فاز اجرا شدند. برای مثال:
	\begin{itemize}
		\item 
ثبت سفارش: فرآیندهایی تعریف شد تا فروشندگان بتوانند سفارشات را سریع وارد سیستم کنند و سیستم به طور خودکار موجودی را چک کند.
		\item 
مدیریت انبار: مکانیسم‌هایی برای ورود و خروج کالا پیاده‌سازی شد تا موجودی همیشه دقیق باشد و از اشتباهات دستی جلوگیری شود.
		\item 
گزارش‌گیری: ابزارهایی برای تولید گزارش‌های فروش و هزینه‌ها ساخته شد تا مدیریت بتواند تصمیم‌گیری‌های سریع‌تری بگیرد.
	\end{itemize}
	\item 
	اجرای نیازمندی‌های غیرکارکردی
	
	ویژگی‌های کیفی سیستم نیز در این فاز مورد توجه قرار گرفت:
	\begin{itemize}
		\item 
سرعت و کارایی: سیستم طوری ساخته شد که عملیات روزانه مانند جستجوی محصول در کمتر از چند ثانیه انجام شود، که این امر بهینه‌سازی عملیات شرکت ماهکس را افزایش می‌دهد.
		\item 
امنیت و دسترسی: سطوح دسترسی متفاوت برای کاربران (فروش، انبار، مدیریت) تعریف شد تا اطلاعات حساس محافظت شود.
		\item 
قابلیت استفاده: تمرکز روی سادگی فرآیندها بود تا کارکنان بدون نیاز به آموزش پیچیده بتوانند از سیستم استفاده کنند.
	\end{itemize}
	\item 
تست اولیه و اصلاحات

پس از ساخت بخش‌های اصلی، تست‌های اولیه انجام شد تا مشکلات احتمالی شناسایی شوند. این شامل بررسی سناریوهای واقعی مانند ثبت سفارش در حجم بالا یا مدیریت کمبود موجودی بود. اصلاحات بر اساس بازخوردهای داخلی اعمال شد تا سیستم پایدارتر شود و با اهداف بهینه‌سازی شرکت ماهکس (مانند کاهش هزینه‌های عملیاتی) همخوانی داشته باشد.

\item 
مدیریت ریسک در ساخت

ریسک‌های احتمالی مانند تأخیر در اجرا یا ناسازگاری بخش‌ها شناسایی و مدیریت شدند. برای مثال، از روش‌های پشتیبان‌گیری منظم استفاده شد تا در صورت بروز مشکل، پیشرفت کار از دست نرود. این رویکرد اطمینان داد که ساخت سیستم بدون اختلال عمده پیش برود.
\end{enumerate}
\subsection{ابزار ها و زبان های مورد نیاز برای توسعه و بهینه سازی پروژه}

پروژه موردنظر مربوط به یک شرکت نرم‌افزاری بسیار بزرگ است که به‌صورت عملیاتی در حال سرویس‌دهی می‌باشد. در چنین سازمان‌هایی معمولاً سیستم‌ها توزیع‌شده (\lr{Distributed}) هستند:
\begin{itemize}
	\item حجم داده و کاربران بالاست
	\item نیاز به مقیاس‌پذیری، پایداری، و بهینه‌سازی هزینه و عملکرد دارند
	\end{itemize}
بنابراین انتخاب ابزارها باید بر اساس موارد زیر انجام شود.
\begin{itemize}
	\item سازگاری با سیستم‌های موجود
	\item
امکان توسعه تدریجی (\lr{Incremental Improvement})
	\item 
	کاهش ریسک مهاجرت
	\item 
	افزایش کارایی و قابلیت نگه‌داری
\end{itemize}
با توجه به موجود بودن بستر برای پلتفرم بیان شده لذا ما برای شروع کار از ابزار های زیر استفاده خواهیم که در حال حاضر نیز بستر شرکت روی این موارد می باشد.
\begin{enumerate}
	\item 
\textbf{\lr{Backend} (لایه منطق کسب‌وکار)}

\section{ فاز ۴: تحویل به مشتری و نگهداری}
\begin{enumerate}
	\item 
آماده‌سازی برای تحویل

در این فاز، سیستم آماده تحویل به شرکت ماهکس شد. ابتدا آموزش‌های لازم برای کاربران نهایی (کارکنان و مدیریت) برنامه‌ریزی شد تا آنها بتوانند سیستم را به راحتی استفاده کنند. همچنین، اسناد راهنما تهیه شد که شامل دستورالعمل‌های ساده برای عملیات روزانه مانند ثبت سفارش یا تولید گزارش بود..
	
	
	\item اجرای نیازمندی‌های کارکردی
	
فرآیند تحویل و راه‌اندازی

تحویل سیستم به صورت مرحله‌ای انجام شد:

	\begin{itemize}
		\item 
راه‌اندازی اولیه: سیستم در محیط واقعی شرکت ماهکس نصب و اجرا شد، با تمرکز روی انتقال داده‌های قدیمی (مانند لیست موجودی) به سیستم جدید بدون اختلال در عملیات روزانه.
		\item 
تست نهایی: پس از راه‌اندازی، تست‌های عملی با داده‌های واقعی انجام شد تا اطمینان حاصل شود که سیستم بهینه‌سازی مورد نظر (مانند کاهش زمان پردازش سفارشات) را فراهم می‌کند.
		\item 
پذیرش مشتری: بازخورد از شرکت ماهکس جمع‌آوری شد و تنظیمات نهایی اعمال گردید تا سیستم کاملاً با نیازهای آنها همخوانی داشته باشد.
	\end{itemize}
	\item 
برنامه نگهداری بلندمدت

برای نگهداری سیستم، برنامه‌ای جامع تدوین شد:
	



	\begin{itemize}
		\item 
	نگهداری پیشگیرانه: بررسی‌های منظم (مثلاً ماهانه) برای جلوگیری از مشکلات آینده، مانند به‌روزرسانی فرآیندها بر اساس تغییرات بازار.
		\item 
	نگهداری اصلاحی: مکانیسم‌هایی برای رفع سریع مشکلات گزارش‌شده توسط کاربران، مانند خطاهای احتمالی در گزارش‌گیری.
		\item 
	نگهداری تطبیقی: امکان تغییر سیستم در آینده برای اضافه کردن ویژگی‌های جدید، مانند پشتیبانی از فروش آنلاین اگر شرکت ماهکس گسترش یابد.
	\end{itemize}
	\item 
ارزیابی عملکرد پس از تحویل

پس از تحویل، عملکرد سیستم ارزیابی شد تا میزان موفقیت بهینه‌سازی سنجیده شود. معیارهایی مانند کاهش زمان عملیات (از دستی به خودکار) و افزایش دقت موجودی بررسی شدند. گزارش‌های ارزیابی نشان داد که سیستم هزینه‌های شرکت ماهکس را تا ۲۰\% کاهش داده است (بر اساس تخمین‌های اولیه).
	
	\item 
مدیریت تغییرات و پشتیبانی

فرآیندهایی برای مدیریت تغییرات آینده تعریف شد، مانند درخواست‌های جدید از سوی مشتری. تیم پشتیبانی مشخص شد تا در صورت نیاز، کمک فوری ارائه دهد و سیستم همیشه کارآمد باقی بماند. این رویکرد تضمین می‌کند که سیستم نه تنها تحویل شود، بلکه در بلندمدت ارزش افزوده ایجاد کند.
\end{enumerate}

\section{تست}
تست سیستم به صورت کامل و در چندین مرحله در پروژه شرکت ماهکس انجام می شود و یکی از مهم‌ترین بخش‌های فاز ساخت و تحویل می باشد.

تست سیستم برای تضمین کیفیت، کارایی، دقت و قابلیت اطمینان سیستم جدید ضروری است. در شرکت ماهکس که فرآیندهای فروش، انبار و مدیریت به صورت دستی و با خطاهای زیاد انجام می‌شد، تست سیستم اهداف زیر را دنبال کرد:
\begin{enumerate}
	\item 
جلوگیری از خطاهای عملیاتی: مثلاً جلوگیری از ثبت اشتباه سفارش یا محاسبه نادرست موجودی.
\item 
اطمینان از تطابق با نیازمندی‌ها: بررسی اینکه سیستم دقیقاً همان چیزی را ارائه می‌دهد که در فاز تحلیل (ثبت سفارش سریع، گزارش‌گیری دقیق، هشدار کمبود) خواسته شده بود.
\item 
کاهش ریسک شکست سیستم: اگر سیستم بدون تست راه‌اندازی شود، ممکن است در زمان واقعی (مثلاً در اوج فروش) از کار بیفتد و خسارت مالی و اعتباری ایجاد کند.
\item 
افزایش اعتماد مشتری (شرکت ماهکس): مدیریت و کارکنان باید مطمئن باشند که سیستم جدید بهتر از روش دستی است.
\item 
بهینه‌سازی واقعی: تست نشان می‌دهد که آیا زمان ثبت سفارش واقعاً از ۵ دقیقه به ۳۰ ثانیه کاهش یافته یا خیر.
\item 
پشتیبانی از نگهداری آینده: مشکلات شناسایی‌شده در تست، راهنمای خوبی برای نگهداری و به‌روزرسانی سیستم هستند.
\end{enumerate}

\subsection{چطور تست انجام شد؟ }
تست سیستم در سه مرحله اصلی و با روش‌های منظم و قابل تکرار انجام شد:
\begin{enumerate}
	\item 
 تست واحد \lr{(Unit Testing)} - در فاز ساخت


هر بخش از سیستم به صورت جداگانه تست شد.
مثال:

تست ثبت سفارش: آیا سیستم کد محصول، تعداد و قیمت را درست دریافت و ذخیره می‌کند؟
تست محاسبه موجودی: آیا پس از خروج کالا، موجودی به‌درستی کم می‌شود؟


هدف: اطمینان از عملکرد صحیح هر بخش قبل از اتصال به بخش‌های دیگر.

\item 
 تست یکپارچگی \lr{(Integration Testing)} - در اواخر فاز ساخت


بخش‌های مختلف سیستم (مثل فروش + انبار + گزارش‌گیری) با هم تست شدند.
مثال:

وقتی سفارش ثبت می‌شود، آیا موجودی انبار به‌صورت خودکار کم می‌شود؟
آیا فاکتور تولیدشده با اطلاعات سفارش و موجودی مطابقت دارد؟


هدف: اطمینان از همکاری صحیح بخش‌ها و جلوگیری از خطاهای ارتباطی.

\item 
 تست پذیرش کاربر\lr{ (User Acceptance Testing - UAT)} - در فاز تحویل


این تست توسط کاربران واقعی شرکت ماهکس (فروشنده، انباردار، مدیر) انجام شد.
سناریوهای واقعی شبیه‌سازی شد:

ثبت ۵۰ سفارش در یک ساعت.
ورود ۱۰۰ کالا به انبار و بررسی گزارش موجودی.
تولید گزارش فروش ماهانه.


کاربران فرم‌های بازخورد پر کردند و گفتند سیستم چقدر ساده، سریع و دقیق است.
هدف: تأیید نهایی اینکه سیستم در محیط واقعی کار می‌کند و نیازهای عملیاتی را برآورده می‌کند.

\item 
 تست عملکرد (\lr{Performance Testing}) - در هر دو فاز


بررسی سرعت، پایداری و ظرفیت سیستم.


\item 
 تست امنیتی (\lr{Security Testing})


بررسی دسترسی غیرمجاز.
مثال: آیا کارمند فروش می‌تواند گزارش مالی مدیریت را ببیند؟
\end{enumerate}

\chapter{استخدام}

اهداف اصلی استخدام:
\begin{itemize}
	\item 
	پوشش تخصص‌های جدید: ایده‌های نو پروژه (مانند تحلیل داده برای بهینه‌سازی هزینه‌ها) نیاز به مهارت‌های تخصصی دارد که ممکن است در تیم فعلی موجود نباشد.
	\item 
	مدیریت مقیاس بزرگ: پروژه لارج اسکیل است، بنابراین نیاز به نیروی بیشتر برای سرعت بخشیدن به فازهای ساخت و نگهداری.
	\item 
	بهینه‌سازی بلندمدت: استخدام کمک می‌کند تا سیستم پایدار بماند و ایده‌های آینده (مانند گسترش به بازارهای جدید) پشتیبانی شوند.
	\item 
	کاهش بار کاری: جلوگیری از خستگی تیم فعلی و افزایش بهره‌وری در بهینه‌سازی عملیات شرکت ماهکس.
	\item 
	نوآوری: نیروی جدید می‌تواند دیدگاه‌های تازه بیاورد و ایده‌های پروژه را غنی‌تر کند.
\end{itemize}

استخدام باید استراتژیک، مبتنی بر نیازهای پروژه و با تمرکز روی کیفیت باشد. رویکرد شامل تحلیل نیازها، جستجوی هدفمند و ادغام سریع نیروها است. برای پروژه بزرگ مانند این، استخدام به صورت مرحله‌ای (ابتدا برای فاز ساخت، سپس نگهداری) انجام شود تا با پیشرفت پروژه همخوانی داشته باشد. اولویت با افرادی است که تجربه در بهینه‌سازی سیستم‌های مشابه داشته باشند و بتوانند ایده‌های جدید را سریع پیاده کنند.


تعداد نیروها بسته به مقیاس پروژه (لارج اسکیل) تخمین زده می‌شود. برای شرکت ماهکس، با توجه به حجم پروژه حدود ۲۵ نفر استخدام شوند، که به صورت زیر توزیع شوند:

	تعداد کلی: ۲۵ نفر (برای پوشش فازهای ساخت، تست و نگهداری).
چطوری (توزیع):
\begin{itemize}
	\item
	۴ نفر در مرحله اولیه .
	\item 
	۵ نفر در مرحله تست و تحویل .
	\item 
	۳ نفر برای نگهداری بلندمدت .
\item 
۶ نفر برای بک اند .
\item 
۲ نفر برای \lr{UX/UI}.

\item 
۱ نفر برای دیتابیس.
\item 
۳ نفر برای دواپس.

\end{itemize}
این تعداد حداقل افرادی است که می توانند در این پروژه سهیم باشند، و ممکن است در طی مسیر نیاز به افراد متخصص بیشتری احساس شود.

\end{enumerate}

\chapter{انتخاب مدل توسعه نرم افزار}

شرکت سریع ترابر ماهان (ماهکس) به عنوان یکی از بازیگران کلیدی صنعت لجستیک و حمل‌ونقل هوایی/زمینی در کشور، با حجم عظیمی از داده‌های عملیاتی روبروست. این داده‌ها شامل اطلاعات مرسولات، وضعیت لحظه‌ای ناوگان، اطلاعات مشتریان و تراکنش‌های مالی است. در حال حاضر، هسته مرکزی سیستم نرم‌افزاری این شرکت بر پایه زبان برنامه‌نویسی\lr{Java Enterprise } بنا شده است. این سیستم طی سال‌ها توسعه یافته و از پایداری خوبی برخوردار است، اما با چالش‌هایی در زمینه مقیاس‌پذیری آنی (\lr{Real-time Scalability}) و سرعت پردازش در زمان‌های اوج ترافیک مواجه است.

\section{هدف پروژه و چالش فنی}

هدف اصلی این پروژه، نوسازی و ارتقای سیستم موجود است، نه جایگزینی کامل آن. استراتژی فنی اتخاذ شده، استفاده از زبان برنامه‌نویسی \lr{Go} (\lr{Golang}) در کنار سیستم جاوای فعلی است. زبان Go به دلیل ماهیت کامپایلی، مدیریت حافظه کارآمد و مدل همزمانی فوق‌العاده \lr{(Goroutines)}، گزینه‌ای ایده‌آل برای توسعه میکروسرویس‌هایی است که نیاز به سرعت بالا و پاسخگویی به تعداد زیادی درخواست همزمان دارند (مانند سرویس رهگیری مرسوله یا سرویس‌های مرتبط با اینترنت اشیاء در انبارها).

چالش اصلی در اینجا، مهاجرت تدریجی و همزیستی مسالمت‌آمیز  این دو تکنولوژی است. ما نمی‌توانیم سیستم فعلی ماهکس را متوقف کنیم؛ بنابراین، انتخاب مدل فرآیند توسعه نرم‌افزار (\lr{SDLC}) حیاتی‌ترین تصمیم در این مرحله است.

\section{ تحلیل و انتخاب مدل فرآیند توسعه نرم‌افزار}

در مهندسی نرم‌افزار، مدل‌های گوناگونی برای مدیریت چرخه حیات تولید وجود دارد. برای پروژه ماهکس، با توجه به شرایط خاص آن (وجود سیستم لگاسی و ورود تکنولوژی جدید)، مدل‌های زیر بررسی شدند:

\subsection{ بررسی مدل‌های رد شده}
\begin{enumerate}
	\item 
مدل آبشاری (\lr{Waterfall}): 

این مدل به دلیل خطی بودن و عدم انعطاف‌پذیری، بدترین گزینه برای ماهکس است. در مدل آبشاری، تمام نیازمندی‌ها باید در ابتدا مشخص باشد. در پروژه ما که قرار است زبان Go به تدریج وارد شود و ممکن است در میانه راه با چالش‌های پیش‌بینی نشده در ارتباط با جاوا مواجه شویم، مدل آبشاری ریسک شکست را به شدت بالا می‌برد.
\item 
مدل نمونه‌سازی \lr{(Prototyping)}:

 اگرچه ساخت نمونه اولیه برای تست ماژول‌های \lr{Go} مفید است، اما این مدل به تنهایی برای کل پروژه کافی نیست. سیستم ماهکس یک سیستم عملیاتی حیاتی \lr{(Mission Critical}) است و نمی‌توان کل آن را بر اساس یک پروتوتایپ دور انداختنی بنا کرد.
\item 
توسعه سریع (\lr{RAD}) و افزایشی \lr{(Incremental)}: 

این مدل‌ها سرعت بالایی دارند اما تمرکز اصلی آنها بر روی مدیریت ریسک‌های فنی پیچیده نیست. پروژه ماهکس بیش از آنکه به سرعت نیاز داشته باشد، به مدیریت ریسک ناشی از ترکیب دو زبان مختلف نیاز دارد.
\item 
مدل منتخب: مدل حلزونی (\lr{Spiral Model})

با توجه به تحلیل فوق، مدل حلزونی به عنوان مناسب‌ترین مدل برای پروژه نوسازی سیستم ماهکس انتخاب می‌شود. این مدل، ویژگی‌های مثبت مدل‌های تکرارشونده و نمونه‌سازی را با تمرکز شدید بر "مدیریت ریسک" ترکیب می‌کند.
\end{enumerate}
\subsection{انتخاب مدل حلزونی}
دلایل کلیدی انتخاب مدل حلزونی برای ماهکس:
\begin{itemize}
	\item 
مدیریت ریسک محور \lr{(Risk-Driven Approach)}:

 بزرگترین نگرانی ما در ماهکس، عدم سازگاری احتمالی سرویس‌های جدید \lr{Go} با دیتابیس‌ها یا سرویس‌های قدیمی جاوا است. در مدل حلزونی، در ابتدای هر چرخه، صراحتاً ریسک‌ها تحلیل می‌شوند. اگر ریسک پیاده‌سازی یک ماژول با \lr{Go} خیلی بالا باشد، می‌توان در همان مرحله تصمیم دیگری گرفت، قبل از اینکه هزینه‌های سنگین کدنویسی تحمیل شود.
\item 
توسعه تکاملی: 

سیستم ماهکس بزرگ است. مدل حلزونی اجازه می‌دهد سیستم را به صورت چرخشی و بخش به بخش (مثلاً ابتدا بخش رهگیری، سپس بخش قیمت‌گذاری و...) به تکنولوژی جدید منتقل کنیم.
\item 
انعطاف در برابر تغییر نیازمندی‌ها: 

صنعت لجستیک پویا است. ممکن است در میانه پروژه، نیاز به افزودن یک سرویس جدید برای پهپادهای تحویل‌دهنده ایجاد شود. مدل حلزونی در هر دور چرخش، اجازه بازبینی نیازمندی‌ها را می‌دهد.
\end{itemize}
 \section{پیاده‌سازی مدل حلزونی در ماهکس}
 
یک چرخه نمونه از مدل حلزونی برای توسعه سرویس محاسبه نرخ حمل‌ونقل با زبان \lr{Go} به شکل زیر خواهد بود:
\begin{itemize}
	\item 
برنامه‌ریزی (\lr{Planning}):

 هدف این چرخه تعیین می‌شود: جدا کردن منطق محاسبه قیمت از کد یکپارچه جاوا و تبدیل آن به یک میکروسرویس \lr{Go}.
\item 
تحلیل ریسک (\lr{Risk Analysis}): 

بررسی می‌کنیم آیا سرویس جدید \lr{Go} می‌تواند با سرعت کافی اطلاعات تعرفه‌ها را از دیتابیس قدیمی \lr{Oracle} که توسط جاوا مدیریت می‌شود، بخواند؟ اگر ریسک کندی بالا باشد، شاید تصمیم بگیریم یک کش (\lr{Cache}) لایه میانی (\lr{Redis}) ایجاد کنیم.
\item 
مهندسی و اجرا (\lr{Engineering}):

 تیم توسعه، میکروسرویس محاسبه نرخ را با زبان \lr{Go} پیاده‌سازی می‌کند. \lr{API}های ارتباطی (مثلاً \lr{gRPC} یا \lr{REST}) برای صحبت با سیستم جاوایی تعریف و پیاده‌سازی می‌شوند. تست‌های واحد و یکپارچگی انجام می‌شود.
\item 
ارزیابی مشتری (\lr{Evaluation}):

 نسخه اولیه در یک محیط محدود (مثلاً فقط برای مرسولات درون شهری تهران) زیر بار می‌رود. عملکرد سیستم رصد شده و بازخوردها برای چرخه بعدی جمع‌آوری می‌شود.
\end{itemize}
\section{تعیین فرآیندهای مهندسی }

مدل بلوغ توانایی  چارچوبی است برای سنجش میزان بلوغ فرآیندهای نرم‌افزاری یک سازمان. برای اینکه پروژه ماهکس از یک تلاش فردی به یک فرآیند سازمانی پایدار تبدیل شود، باید وضعیت فعلی و هدف آینده را در قالب این سطوح پنج‌گانه تعریف کنیم.
\begin{enumerate}
	\item 
سطح آغازین (\lr{Initial})

در این سطح، فرآیندها مستند نشده و غیرقابل پیش‌بینی هستند. موفقیت پروژه وابسته به قهرمانان فردی است.

وضعیت در ماهکس: اگر در حال حاضر تنها یکی دو نفر از برنامه‌نویسان ارشد دقیقاً می‌دانند که کدهای قدیمی جاوا چگونه کار می‌کنند و هیچ مستنداتی وجود ندارد، ماهکس در سطح ۱ است. اگر ورود زبان Go بدون هیچ استاندارد مشخصی و صرفاً بر اساس سلیقه برنامه‌نویس جدید انجام شود، ما همچنان در این سطح خواهیم ماند که بسیار خطرناک است.
\item 
سطح ۲: تکرارپذیر (\lr{Repeatable}) 

در این سطح، فرآیندهای مدیریت پروژه اولیه استقرار یافته‌اند. موفقیت‌های قبلی قابل تکرار هستند.

پیاده‌سازی در ماهکس: 

ما باید برای توسعه ماژول‌های \lr{Go}، تخمین زمان و هزینه داشته باشیم. باید از ابزارهای کنترل نسخه (مانند \lr{Git}) به شکل ساختارمند استفاده کنیم. اگر تیم موفق شد ماژول رهگیری  را با \lr{Go }بازنویسی کند، باید بتواند از همان تجربیات مدیریتی برای بازنویسی ماژول مدیریت رانندگان استفاده کند. در این سطح، ما دیگر به حافظه افراد متکی نیستیم، بلکه به فرآیندهای مدیریت پروژه متکی هستیم.
\item 
 تعریف شده (\lr{Defined}) 

در این سطح، فرآیندهای مهندسی و مدیریتی مستند شده، استاندارد شده و در کل سازمان یکپارچه شده‌اند. این سطح برای پروژه ترکیبی جاوا/گو حیاتی است.


پیاده‌سازی در ماهکس: باید سند استاندارد توسعه نرم‌افزار ماهکس تدوین شود. در این سند صراحتاً تعریف می‌شود که:

کدام دسته از وظایف باید با \lr{Java} بمانند (مثلاً تراکنش‌های مالی حساس).

کدام دسته از وظایف باید با \lr{Go} توسعه یابند (مثلاً پردازش‌های سریع و بلادرنگ).

پروتکل ارتباطی استاندارد بین سرویس‌های جاوا و گو چیست (مثلاً استفاده اجباری از \lr{Protocol Buffers} و \lr{gRPC} برای ارتباطات داخلی). در این سطح، همه تیم‌ها از یک "زبان مشترک" در توسعه پیروی می‌کنند.
\item 
مدیریت شده (\lr{Managed}) 
در این سطح، سازمان با استفاده از داده‌ها و معیارهای دقیق کمی (\lr{Metrics})، فرآیندها را کنترل می‌کند. ما دیگر حدس نمی‌زنیم، بلکه اندازه‌گیری می‌کنیم.

پیاده‌سازی در ماهکس: برای سیستم جدید، ما شروع به جمع‌آوری متریک‌ها می‌کنیم:

متوسط زمان پاسخگویی (\lr{Latency}) سرویس جدید \lr{Go} در مقایسه با ماژول قدیمی جاوا چند میلی‌ثانیه بهبود یافته است؟

نرخ خطا (\lr{Error Rate}) در ارتباط بین دو زبان چقدر است؟

میزان مصرف منابع سرور (\lr{CPU/RAM}) پس از مهاجرت به \lr{Go} چقدر کاهش یافته است؟ مدیران ماهکس بر اساس این اعداد تصمیم می‌گیرند که آیا به مهاجرت ادامه دهند یا استراتژی را تغییر دهند.
\item 
بهینه شده (\lr{Optimized}) 
این بالاترین سطح بلوغ است که در آن سازمان به طور مداوم فرآیندهای خود را بر اساس بازخوردها بهبود می‌بخشد.

پیاده‌سازی در ماهکس: تیم فنی ماهکس به طور فعال به دنبال تکنولوژی‌های جدیدتر در اکوسیستم \lr{Go} یا \lr{Java} می‌گردد تا کارایی را افزایش دهد. مثلاً بررسی می‌کنند که آیا استفاده از فریم‌ورک‌های جدیدتر برای مدیریت میکروسرویس‌ها می‌تواند هزینه‌های عملیاتی را ۵\% دیگر کاهش دهد؟ تمرکز بر پیشگیری از نقص‌هاست تا کشف آن‌ها.
\end{enumerate}
\section{نتیجه‌گیری و چشم‌انداز معماری}
انتخاب مدل حلزونی به عنوان متدولوژی توسعه، به شرکت ماهکس این امکان را می‌دهد که با اطمینان خاطر و کمترین ریسک، از یک سیستم یکپارچه (\lr{Monolithic}) جاوایی به سمت یک معماری مدرن مبتنی بر میکروسرویس‌ها (ترکیبی از \lr{Java} و \lr{Go}) حرکت کند.

این تغییر معماری، با استفاده از زبان \lr{Go} در لبه‌های بیرونی سیستم (که با مشتری در ارتباط است) باعث افزایش سرعت و رضایت مشتریان می‌شود، در حالی که هسته پایدار جاوایی همچنان امنیت تراکنش‌ها را تضمین می‌کند.

\chapter{رمزنگاری}
\section{ضرورت استفاده از مکانیزم‌های رمزنگاری و اعتبارسنجی در پروژه}
در سیستم‌های نرم‌افزاری که با اطلاعات حساس و هویتی سروکار دارند، صرفاً ذخیره‌سازی داده‌ها به‌صورت ساده و بدون کنترل امنیتی، ریسک بالایی ایجاد می‌کند. یکی از مهم‌ترین چالش‌ها در این نوع سیستم‌ها، جلوگیری از ثبت اطلاعات نادرست، جعلی یا دستکاری‌شده و همچنین حفظ محرمانگی داده‌ها در پایگاه داده است.

به همین منظور، در این پروژه علاوه بر استفاده از مکانیزم‌های رمزنگاری، از روش‌های اعتبارسنجی داده نیز استفاده شده است. این کار باعث می‌شود اطلاعات ذخیره‌شده هم از نظر امنیتی محافظت شوند و هم از نظر صحت و یکپارچگی، قابل اعتماد باشند. ترکیب این دو رویکرد نقش مهمی در افزایش کیفیت، امنیت و پایداری سیستم ایفا می‌کند.

\section{استفاده از الگوریتم رقم کنترل مضرب ۱۱ برای کد پرسنلی}

کد پرسنلی به‌عنوان یکی از مهم‌ترین شناسه‌های هر فرد در سیستم، نیازمند دقت و کنترل بالایی در زمان ثبت و نگهداری است. در این پروژه، برای افزایش صحت اطلاعات و جلوگیری از ورود کدهای پرسنلی اشتباه یا غیرمعتبر، از الگوریتم محاسبه رقم کنترل بر اساس مضرب ۱۱ استفاده شده است.

در این روش، ارقام تشکیل‌دهنده کد پرسنلی با ضرایب مشخصی ضرب شده و مجموع حاصل‌ضرب‌ها محاسبه می‌شود. سپس باقی‌مانده این مجموع بر عدد ۱۱ به‌عنوان رقم کنترل در نظر گرفته می‌شود. در زمان ثبت اطلاعات، سیستم با محاسبه مجدد رقم کنترل و مقایسه آن با رقم کنترل موجود در کد پرسنلی، معتبر بودن یا نبودن کد را بررسی می‌کند.

استفاده از الگوریتم مضرب ۱۱ باعث می‌شود خطاهای رایج در ورود داده‌ها، مانند جابه‌جایی ارقام یا وارد کردن اشتباه یک رقم، به‌راحتی شناسایی شوند. این مکانیزم در کنار رمزنگاری داده‌ها، سطح اطمینان و امنیت اطلاعات پرسنلی ذخیره‌شده در پایگاه داده را به‌طور قابل توجهی افزایش می‌دهد.

\section{  محاسبه رقم کنترل در کد پرسنلی با روش تصاعد عددی}

با توجه به تعداد بالای نیروی انسانی در سازمان، در این پروژه از کد پرسنلی ۹ رقمی برای شناسایی یکتای هر فرد استفاده شده است. این ساختار به‌گونه‌ای طراحی شده که علاوه بر پوشش تعداد بالای پرسنل، امکان دسته‌بندی و اعتبارسنجی اطلاعات را نیز فراهم کند.

ساختار کد پرسنلی به شکل زیر تعریف شده است:

۵ رقم ابتدایی: شناسه یکتای فرد در سیستم

۳ رقم میانی: کد مربوط به دپارتمان یا واحد سازمانی

۱ رقم انتهایی: رقم کنترل \lr{(Check Digit)}

سه رقم میانی این امکان را فراهم می‌کند که هر دپارتمان یا واحد سازمانی دارای یک بازه مشخص از کدها باشد و از این طریق مدیریت و گزارش‌گیری اطلاعات پرسنل ساده‌تر و منظم‌تر انجام شود. رقم انتهایی نیز به‌عنوان رقم کنترل، برای اعتبارسنجی کل کد پرسنلی مورد استفاده قرار می‌گیرد.

محاسبه رقم کنترل با الگوریتم مضرب ۱۱

برای محاسبه رقم کنترل، از الگوریتم مضرب ۱۱ استفاده شده است. در این روش، ابتدا ۸ رقم اول کد پرسنلی در نظر گرفته می‌شود و هر رقم در یک ضریب مشخص ضرب می‌گردد. سپس مجموع حاصل‌ضرب‌ها محاسبه شده و باقی‌مانده تقسیم این مجموع بر عدد ۱۱ به‌عنوان رقم کنترل در نظر گرفته می‌شود.

مثال محاسبه:

فرض می‌کنیم ۸ رقم اول کد پرسنلی به صورت زیر باشد:

\begin{center}
	\lr{12345678}
\end{center}

ضرایب از چپ به راست به‌صورت 8 تا 1 در نظر گرفته می‌شوند:
\begin{center}
	\begin{latin}
		1 × 8 = 8
		
		2 × 7 = 14
		
		3 × 6 = 18
		
		4 × 5 = 20
		
		5 × 4 = 20
		
		6 × 3 = 18
		
		7 × 2 = 14
		
		8 × 1 = 8
		
	\end{latin}
\end{center}
مجموع حاصل‌ضرب‌ها برابر است با:
\begin{center}
\lr{120}
\end{center}


حال عدد 120 از  121  کم می‌شود:
\begin{center}
\lr{121 - 120 = 1 }
\end{center}

در نتیجه، رقم کنترل برابر با 1 خواهد بود.

بنابراین کد پرسنلی کامل به شکل زیر خواهد بود:

\begin{center}
	\lr{123456781}
\end{center}

استفاده از این ساختار ۹ رقمی به همراه الگوریتم رقم کنترل مضرب ۱۱، باعث افزایش دقت در ثبت اطلاعات، جلوگیری از ورود کدهای پرسنلی اشتباه و حفظ یکپارچگی داده‌ها در پایگاه داده می‌شود. این مکانیزم در کنار رمزنگاری اطلاعات، نقش مهمی در افزایش امنیت و قابلیت اطمینان سیستم ایفا می‌کند.

\chapter{ریسک}
مدیریت ریسک فرایندی است که طی آن تهدیدهای احتمالی شناسایی، ارزیابی و برای مقابله با آن‌ها برنامه‌ریزی می‌شود. در پروژه شرکت سریع ترابر ماهان، هدف ما تضمین پایداری سیستم در ارائه خدمات لجستیکی است.
\begin{enumerate}
	\item 
 شناسایی ریسک‌ها \lr{(Risk Identification)}
 
در اولین قدم، ریسک‌هایی که ممکن است موفقیت پروژه را تهدید کنند شناسایی می‌کنیم. این ریسک‌ها در سه حوزه اصلی قرار می‌گیرند:

ریسک‌های فنی: مربوط به تکنولوژی‌های مورد استفاده، پایگاه داده و زیرساخت‌های شبکه.

ریسک‌های انسانی: مربوط به خطای کاربران، عدم پذیرش سیستم توسط رانندگان یا پرسنل و کمبود تخصص.

ریسک‌های محیطی/سازمانی: تغییر قوانین حمل‌ونقل، نوسانات اقتصادی و حوادث پیش‌بینی نشده در جاده‌ها.
\item 
 تحلیل و دسته‌بندی ریسک‌ها
 
در این بخش، ریسک‌ها را به دو دسته اصلی نرم‌افزاری و عملیاتی تقسیم می‌کنیم تا مدیریت آن‌ها ساده‌تر شود.
\begin{itemize}
	\item 
تحلیل ریسک‌های نرم‌افزاری

این ریسک‌ها مستقیماً با فرآیند تولید و نگهداری نرم‌افزار شرکت در ارتباط هستند:
\begin{itemize}
	\item 
ناپایداری سیستم در زمان اوج ترافیک: احتمال کند شدن یا قطعی سرور در زمان‌هایی که تعداد درخواست‌های ثبت بار یا رهگیری محموله زیاد است.
\item 
نقص در امنیت داده‌ها: با توجه به اینکه اطلاعات مشتریان و محموله‌های باارزش در سیستم ثبت می‌شود، ریسک نفوذ و سرقت داده‌ها یک تهدید جدی است.
\item 
ناسازگاری با سخت‌افزارهای قدیمی: رانندگان ممکن است از گوشی‌های هوشمند قدیمی استفاده کنند و اپلیکیشن روی آن‌ها به درستی اجرا نشود.
\item
خطا در یکپارچه‌سازی (\lr{Integration}): مشکل در اتصال سیستم شرکت به درگاه‌های بانکی یا نقشه‌های گوگل/نشان برای مسیریابی.
\end{itemize}
\item 
ریسک‌های عملیاتی

این ریسک‌ها مربوط به دنیای واقعی و فرآیندهای اجرایی شرکت ماهکس است:
\begin{itemize}
	\item 

اختلال در شبکه \lr{GPS}: از دست رفتن سیگنال در مناطق دورافتاده که منجر به توقف به‌روزرسانی لحظه‌ای مکان محموله می‌شود.
\item 
خطای ورود داده توسط پرسنل انبار: ثبت اشتباه وزن یا نوع کالا که منجر به تخصیص خودروی نامناسب می‌شود.
\item
تاخیر در ارسال به دلیل خرابی ناوگان: ریسک فیزیکی که مستقیماً بر اعتبار نرم‌افزار (در محاسبه زمان رسیدن) تاثیر می‌گذارد.
\end{itemize}
\end{itemize}
\item 
تجزیه ریسک (\lr{Risk Decomposition})

برای درک بهتر، ریسک‌های بزرگ را به اجزای کوچک‌تر تقسیم می‌کنیم. به عنوان مثال، «ریسک عدم پذیرش سیستم توسط کارکنان» به موارد زیر تجزیه می‌شود:
\begin{itemize}
	\item 
سختی رابط کاربری (\lr{UI/UX}) برای افراد غیرفنی.
\item 
ترس کارکنان از جایگزینی سیستم با نیروی انسانی.
\item 
فقدان آموزش کافی برای استفاده از امکانات جدید سیستم.
\item
با این تجزیه، ما متوجه می‌شویم که به جای "تغییر کل سیستم"، باید روی "آموزش" و "سادگی طراحی" تمرکز کنیم.
\end{itemize}
\item 
 برآورد و استراتژی‌های کاهش ریسک
 
برای هر ریسک شناسایی شده، باید برنامه‌ای داشته باشیم. این برنامه شامل دو رویکرد زیر است:
\begin{enumerate}
	\item 
اجتناب از ریسک \lr{(Risk Avoidance)}
در این روش، ما طوری برنامه‌ریزی می‌کنیم که احتمال وقوع ریسک به صفر برسد:
\begin{itemize}
	\item 
استفاده از سرورهای پشتیبان \lr{(Redundancy)}:

 برای جلوگیری از قطعی سیستم، از چند سرور در موقعیت‌های جغرافیایی مختلف استفاده می‌کنیم.
\item 
انتخاب تکنولوژی‌های بالغ: 

به جای استفاده از فریم‌ورک‌های بسیار جدید و تست نشده، از زبان‌های برنامه‌نویسی پایدار (مثل \lr{Java} یا \lr{Go}) استفاده می‌کنیم تا با باگ‌های ناشناخته روبرو نشویم.
\end{itemize}
\item 
 تشخیص و حذف ریسک \lr{(Risk Detection \& Removal)}
 
اگر نتوانیم از ریسکی اجتناب کنیم، باید سیستمی برای تشخیص سریع و حذف آن داشته باشیم:
\begin{itemize}
	\item 
تست‌های دوره‌ای (\lr{QA}):

 انجام تست‌های نفوذ \lr{(Penetration Test) } برای شناسایی حفره‌های امنیتی قبل از اینکه هکرها آن‌ها را پیدا کنند.
\item 
مانیتورینگ لحظه‌ای: 

استفاده از ابزارهایی که به محض افت سرعت سیستم یا قطع شدن دسترسی یک راننده، به تیم پشتیبانی هشدار (\lr{Alert}) می‌دهند.
\end{itemize}
\end{enumerate}
\item 
 محدودسازی خسارات \lr{(Damage Limitation)}

اگر ریسک به وقوع پیوست و مشکلی ایجاد شد، چگونه اثر آن را به حداقل برسانیم؟
\begin{itemize}
	\item 
سیستم آفلاین در اپلیکیشن رانندگان: در صورت قطع اینترنت، اپلیکیشن اطلاعات را در حافظه گوشی ذخیره کرده و به محض اتصال مجدد، با سرور همگام (\lr{Sync}) می‌شود.
\item 
پشتیبان‌گیری (\lr{Backup}) منظم: تهیه نسخه پشتیبان روزانه از پایگاه داده در حافظه‌های جداگانه تا در صورت خرابی سرور، اطلاعات مشتریان از بین نرود.
\item 
طرح جایگزین دستی: پیش‌بینی فرم‌های کاغذی استاندارد برای زمان‌های اضطراری که سیستم کامپیوتری به کلی در دسترس نیست، تا فرایند حمل‌ونقل متوقف نشود.
\end{itemize}
\end{enumerate}

\section{نتیجه‌گیری}
هدف از انجام این پروژه، بررسی، بهبود و بهینه‌سازی سیستم موجود شرکت سریع‌ترابر ماهان بوده است. سیستمی که از زمان تأسیس این شرکت تاکنون بدون تغییر اساسی مورد استفاده قرار گرفته و نیازمند بازنگری و ارتقا متناسب با شرایط و نیازهای روز بوده است. در این پروژه تلاش شد با تحلیل وضعیت فعلی سیستم، شناسایی نقاط ضعف و محدودیت‌ها، و ارائه راهکارهای پیشنهادی، زمینه‌ای برای افزایش کارایی، سرعت انجام فرآیندها و بهبود عملکرد کلی مجموعه فراهم شود.

اجرای این پروژه فرصتی ایجاد کرد تا مفاهیم آموزشی به‌صورت عملی مورد بررسی قرار گرفته و نگاه تحلیلی نسبت به ساختارها و فرآیندهای سازمانی تقویت شود. همچنین نتایج به‌دست‌آمده می‌تواند مبنایی مناسب برای تصمیم‌گیری‌های آتی و اعمال تغییرات مؤثر در سیستم موجود شرکت باشد.
